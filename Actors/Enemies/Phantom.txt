ACTOR Phantom: actor 13098
{
  Health 400
  Radius 18
  Height 56
  Mass 600
  Speed 7
  PainChance 128
  Monster
  damage 2
  +FLOAT
  +NOGRAVITY
  +STEALTH
  +MISSILEMORE
  stealthalpha 0.5
  SeeSound "radiohaunt/see"
  PainSound ""
  DeathSound "radiohaunt/death"
  ActiveSound ""
  AttackSound ""
  Obituary "%o died from cerebral hemorrhage."
  scale 0.8
  States
  {
  Spawn:
    DUKH ABCD 6 A_Look
    Loop
  See:
	DUKH A 0 A_Jump(200,2)
	DUKH A 0 A_StartSound("radiohaunt/idle",CHAN_BODY,CHANF_NOSTOP)//these guys should be a bit more chatty
	DUKH A 0 
    DUKH A 4 A_CHASE
	DUKH A 0 A_SpawnItemEX("DUKHAfterImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	DUKH B 4 A_Chase
	DUKH A 0 A_SpawnItemEX("DUKHAfterImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	DUKH C 4 A_Chase
	DUKH A 0 A_SpawnItemEX("DUKHAfterImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	DUKH D 4 A_FastChase
	DUKH A 0 A_SpawnItemEX("DUKHAfterImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
    Loop
  Melee:
    DUKH E 3 //A_PlaySound("haunt/attack")
	DUKH EF 4 A_FaceTarget
	DUKH F 0 A_CustomMeleeAttack(random(1,6)*2, "haunt/hit")
	DUKH F 0 A_radiusgive("hauntpainsphere",32,RGF_PLAYERS|RGF_MONSTERS)
	DUKH F 30 A_blast (0,32,64)
    goto see
  Missile:
    DUKH E 0 A_jumpifcloser(128,"charge")
    DUKH E 3 //A_PlaySound("haunt/attack")
    DUKH FF 6 A_FaceTarget
    DUKH F 10 A_CustomMissile("Traumaball",48,15)
    Goto See
  Charge:
    DUKH EF 8 A_FaceTarget
	DUKH H 4 A_SkullAttack
	DUKH H 4
	DUKH I 4 A_SpawnItemEX("DUKHChargeImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	DUKH J 4 
    DUKH K 4 A_SpawnItemEX("DUKHChargeImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
    Goto charge+3
  Pain:
    DUKH G 3 A_Pain
    DUKH G 3 
    Goto See
  Death:
    GHST J 1 A_SetSolid
    GHST J 1 A_SpawnItemEX("DUKHDeathImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	GHST J 1 A_Scream 
	GHST J 6 A_FadeIn(0.5)
	GHST J 0 A_SetSolid
	GHST J -1
	stop
  Crash:
    GHST K 8 A_PlaySound ("haunt/fall")
	GHST K 0 A_UnSetShootable
    GHST LM 8 A_FadeIn(0.5)
    GHST N 8 A_NoBlocking
    GHST O 1 A_SetFloorClip
	GHST O -1
    Stop
  BURN:
    GHST P 3 bright A_SetSolid
	GHST Q 3 bright A_SpawnItemEX("DUKHDeathImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	GHST P 3 bright A_SetSolid
	GHST QPQPQPQPQ 3 bright
	GHST PQPQPQPQPQ 4 bright  A_FadeIn(0.5)
	GHST P 3 A_SetSolid
	GHST Q -1
	stop
   crash.fire:
    GHST P 3 bright A_PlaySound("radiohaunt/burn")
	GHST Q 3 bright A_SpawnItemEX("DUKHDeathImage", 0, 0, 0, 0, 0, 0, 0, SXF_TRANSFERPOINTERS)
	GHST P 3 bright A_SetSolid
	GHST QPQPQPQ 3 bright A_FadeIn(0.5)
	GHST RSTUVTUVTUVTUVTUVWX 4 bright
	GHST Y 3 A_SetSolid
	GHST Z -1
	stop
  }
}
